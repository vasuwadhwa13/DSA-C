19.ENTER THE STRING FROM THE USER AND CONVERT INTO A FLOAT VALUE?
#include <stdio.h>

int main() 
{
char str[10];
float result = 0.0, decimalPlace = 1.0;
int i = 0, isNegative = 0;

printf("Enter a string to convert into a float: ");
gets(str);

if (str[0] == '-') {
    isNegative = 1;
    i++;
}

while (str[i] >= '0' && str[i] <= '9') 
{
    result = result * 10 + (str[i] - '0');
    i++;
}

if (str[i] == '.') 
{
    i++;
    while (str[i] >= '0' && str[i] <= '9') 
    {
        decimalPlace *= 0.1;
        result += (str[i] - '0') * decimalPlace;
        i++;
    }
}

if (isNegative)
{
    result = -result;
}

printf("The float value is: %.2f\n", result);

return 0;
}

    {
        sentenceLen++;
    }

        wordLen++;

27.DISPLAY THE STRING ON THE OUTPUT SCREEN BUT ASK THE TEXT ALIGNMENT
    OPTION GIVEN BY THE USER (1 FOR LEFT, 2 FOR CENTER, 3 RIGHT)

#include <stdio.h>
#include <string.h>
    
int main() 
{
    char str[100];
    int alignment;

    printf("Enter a string: ");
    gets(str);

    printf("Choose text alignment (1 for left, 2 for center, 3 for right): ");
    scanf("%d", &alignment);

    int length = strlen(str);
    int width = 50; // Width of the output area

    switch (alignment) 
    {
        case 1: // Left alignment
            printf("%s\n", str);
            break;
        case 2: // Center alignment
            printf("%*s\n", (width + length) / 2, str);
            break;
        case 3: // Right alignment
            printf("%*s\n", width, str);
            break;
        default:
            printf("Invalid option.\n");
            break;
    }

    return 0;
}
    
        
        
OUTPUT

Enter a string: vassy
Choose text alignment (1 for left, 2 for center, 3 for right): 2
                      vassy

#include <stdio.h>
#include <string.h>
    
#define MAX_WORDS 10
#define MAX_LENGTH 30
    
typedef struct {
    char word[MAX_LENGTH];
    char meaning[MAX_LENGTH];
} DictionaryEntry;

void create(DictionaryEntry dict[], int *size) {
    if (*size >= MAX_WORDS) {
        printf("Dictionary is full. Cannot add more words.\n");
        return;
    }
    
    printf("Enter word: ");
    scanf("%s", dict[*size].word);
    printf("Enter meaning: ");
    scanf(" %[^\n]", dict[*size].meaning); // Read string with spaces
    (*size)++;
    printf("Word added successfully!\n");
}

void delete(DictionaryEntry dict[], int *size) {
    char word[MAX_LENGTH];
    printf("Enter word to delete: ");
    scanf("%s", word);
    
    for (int i = 0; i < *size; i++) {
        if (strcmp(dict[i].word, word) == 0) {
            for (int j = i; j < *size - 1; j++) {
                dict[j] = dict[j + 1];
            }
            (*size)--;
            printf("Word deleted successfully!\n");
            return;
        }
    }
    printf("Word not found.\n");
}

void search(DictionaryEntry dict[], int size) {
    char word[MAX_LENGTH];
    printf("Enter word to search: ");
    scanf("%s", word);
    
    for (int i = 0; i < size; i++) {
        if (strcmp(dict[i].word, word) == 0) {
            printf("Meaning of '%s': %s\n", dict[i].word, dict[i].meaning);
            return;
        }
    }
    printf("Word not found.\n");
}

int main() 
{
    DictionaryEntry dict[MAX_WORDS] = {
        {"apple", "A fruit that is typically red or green."},
        {"banana", "A long curved fruit that is yellow when ripe."},
        {"grape", "A small, round fruit that can be purple or green."},
        {"orange", "A citrus fruit that is orange in color."},
        {"kiwi", "A small brown fruit with green flesh."},
        {"mango", "A tropical stone fruit with a sweet taste."},
        {"peach", "A soft, round fruit with fuzzy skin."},
        {"pear", "A sweet fruit with a rounded base and tapering top."},
        {"plum", "A smooth-skinned fruit that can be purple or red."},
        {"pineapple", "A tropical fruit with a spiky exterior."}
    };

    int size = MAX_WORDS;
    int choice;

    do {
        printf("\nDictionary Menu:\n");
        printf("1. Create\n");
        printf("2. Delete\n");
        printf("3. Search\n");
        printf("4. Exit\n");
        printf("Choose an option: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                create(dict, &size);
                break;
            case 2:
                delete(dict, &size);
                break;
            case 3:
                search(dict, size);
                break;
            case 4:
                printf("Exiting...\n");
                break;
            default:
                printf("Invalid option. Please try again.\n");
        }
    } while (choice != 4);

    return 0;
}

 29.ENTER THE STRING FROM USER AND CONVERT CHARACTERS INTO THEIR ASCII
 VALUES IN THE INTEGER ARRAY. [FOR EXAMPLE TAKE 10 CHARACTERS] 

 #include <stdio.h>

int main() 
{
    char inputString[11]; 
    int asciiValues[10];

    printf("Enter a string (up to 10 characters): ");
    gets(inputString);

    for (int i = 0; i < 10 && inputString[i] != '\0'; i++) 
    {
        asciiValues[i] = (int)inputString[i];
    }

    printf("ASCII values of the characters in the string:\n");
    for (int i = 0; i < 10 && asciiValues[i] != 0; i++) 
    {
        printf("%c: %d\n", inputString[i], asciiValues[i]);
    }

    return 0;
}
