#include<stdio.h>
void touppercase (char str[])
{ 
    for(int i=0; str[i]!= '\0'; i++)
    { 
        if(str[i] >= 'a' && str[i] <= 'z' )
        {
            str[i] = str[i] - 32; 
        }
    }
} 

int main()
{ 
    char str[100];
    printf("Enter your string : "); 
    gets(str); 
    printf("String before upper case =  %s \n", str);
    touppercase(str); 
    printf("String after upper case =  %s \n", str);
    return 0;
}


OUTPUT

String before upper case= john
String after upper case= JOHN
